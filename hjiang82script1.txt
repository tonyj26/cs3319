DROP USER 'ta'@'localhost';

SHOW TABLES;
CREATE TABLE product (ProductID CHAR(2), Description VARCHAR(20), CostPerItem DECIMAL(20,2), Quantity INT, PRIMARY KEY (ProductID));
CREATE TABLE agent (AgentID CHAR(2) NOT NULL, FirstName VARCHAR(30), LastName VARCHAR(30), City VARCHAR(15), CommissionPercent DECIMAL(10,10), PRIMARY KEY (AgentID)); 
CREATE TABLE customer (CustomerID CHAR(2), City VARCHAR(15), FirstName VARCHAR(30), LastName VARCHAR(30), PhoneNumber VARCHAR(10), AgentID CHAR(2) NOT NULL, PRIMARY KEY (CustomerID), FOREIGN KEY (AgentID) REFERENCES agent(AgentID)); 
CREATE TABLE purchase (ProductID CHAR(2) NOT NULL, CustomerID CHAR(2) NOT NULL, Quantity INT, PRIMARY KEY(ProductID, CustomerID), FOREIGN KEY (ProductID) REFERENCES product(ProductID), FOREIGN KEY (CustomerID) REFERENCES customer(CustomerID));
